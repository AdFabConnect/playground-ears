 19 pl.ready(function ()
 20 {
 21     'use strict';
 22     
 23 	/**
 24 	 * @namespace reference to Adfab.Playground.App object
 25 	 */
 26 	
 27     /** App Object
 28      * @class
 29      * @name PG.App
 30      */
 31     function App ()
 32     {
 33         return this;
 34     }
 35     
 36     /**
 37      * Start the App object by using init
 38      * This method create the namespace used all over the application
 39      * @function
 40      * 
 41      * @name PG.App.init
 42      * 
 43      * @param {null} no params
 44      * @return {null} no return
 45      * 
 46      * @this {App}
 47      * 
 48      * @ignore
 49      * 
 50      * @since version 1.0.0
 51      */
 52     App.prototype.init = function ()
 53     {
 54         // Namespace
 55         PG = Adfab.Playground; 
 56         
 57         // check api key
 58         if(PG.Util.not_null(PG.Cache.settings) && PG.Util.not_null(PG.Cache.settings.apiKey)) {
 59             PG.User.env = _plgd_settings;
 60             PG.Util.log('APIKEY FOUND = ' + PG.Cache.settings.apiKey);
 61         }else {
 62             PG.Util.log('NOT APIKEY FOUND');
 63             return;
 64         }
 65         
 66         PG.App.easyXDM();
 67         PG.App.bindEvent();
 68         PG.User.init();
 69     };
 70     
 71     /**
 72      * Bind the event to track user move
 73      * @function
 74      * 
 75      * @name PG.App.bindEvent
 76      * 
 77      * @param {null} no params
 78      * @return {null} no return
 79      * 
 80      * @this {App}
 81      * 
 82      * @ignore
 83      * 
 84      * @since version 1.0.0
 85      */
 86     App.prototype.bindEvent = function ()
 87     {
 88         // AJAX
 89         var xhrListn = new Object();
 90             xhrListn.open = PG.Cache.xmlhttp.prototype.open;
 91             xhrListn.send = PG.Cache.xmlhttp.prototype.send;
 92         
 93         // Catch all the xhr open event
 94         PG.Cache.xmlhttp.prototype.open = function (a, b)
 95         {
 96             if (!a) var a='';
 97             if (!b) var b='';
 98             
 99             xhrListn.open.apply(this, arguments);
100             xhrListn.method = a;  
101             xhrListn.url = b;
102             if (a.toLowerCase() == 'get') {
103                 xhrListn.data = b.split('?');
104                 xhrListn.data = xhrListn.data[1];
105             }
106         };
107         
108         // Catch all the xhr send event
109         PG.Cache.xmlhttp.prototype.send = function (a, b)
110         {
111             if (!a) var a='';
112             if (!b) var b='';
113             xhrListn.send.apply(this, arguments);
114             if(xhrListn.method.toLowerCase() == 'post') xhrListn.data = a;
115         };
116         
117         // Catch all the load finished event
118         PG.Cache.document.onreadystatechange = function (e)
119         {
120             if(PG.Util.not_null(e.data) && e.data != '') {
121                 PG.App.send(e.data);
122             }
123         };
124         
125         PG.Cache.window.onbeforeunload = function (e)
126         {
127             PG.User.quit();
128             return null;
129         };
130     };
131     
132     /**
133      * Init easyXDM by calling this method
134      * Using easyXDM as RPC mode
135      * @function
136      * 
137      * @name PG.App.easyXDM
138      * 
139      * @param {null} no params
140      * @return {null} no return
141      * 
142      * @this {App}
143      * 
144      * @ignore
145      * 
146      * @since version 1.0.0
147      */
148     App.prototype.easyXDM = function ()
149     {
150         PG.App.rpc = new easyXDM.Rpc(
151         {
152             // fallback NameTransport
153             local: PG.Cache.protocol + PG.Cache.config.url + "easyXDM/name.html",
154             // fallback swf
155             swf: PG.Cache.protocol + PG.Cache.config.url + "easyXDM/easyxdm.swf",
156             // page to load
157             remote: PG.Cache.protocol + PG.Cache.config.url + "index.html"
158         },
159         {
160             remote: {
161                 request: {}
162             }
163         });
164     };
165     
166     /**
167      * Send a request to easyXDM using RPC with no return
168      * @function
169      * 
170      * @name PG.App.send
171      * 
172      * @param {string} url of the service
173      * @return {Boolean} valid url
174      * 
175      * @this {App}
176      * 
177      * @example
178      * PG.App.send( {String} url)
179      * 
180      * @since version 1.0.0
181      */
182     App.prototype.send = function (url)
183     {
184         var validUrl = PG.Util.isUrlValid(url);
185         if(validUrl) {
186             var userUrl = PG.Cache.protocol + PG.Cache.config.url + PG.Cache.config.send + '?apiKey='
187                 + PG.Cache.settings.apiKey + '&url=' + url + '&uid=' + PG.User.uid;
188             
189             PG.Util.log('send url : ' + userUrl);
190             
191             if(PG.User.isLogged()) {
192                 userUrl += '&login=' + PG.User.id;
193             }
194             
195             PG.App.rpc.request(
196                 { 
197                     url: userUrl
198                 },
199                 function (rpcdata)
200                 {
201                     PG.Util.log(rpcdata);
202                 },
203                 function (error)
204                 {
205                     throw 'Api Error, code : ' + error.code + ', msg : ' + error.message;
206                 }
207             );
208         }
209         return validUrl;
210     }; 
211     
212     /**
213      * Send a request to easyXDM using RPC waiting for a json return
214      * @function
215      * 
216      * @name PG.App.call
217      * 
218      * @param {string} serviceName
219      * @return {Object} promise
220      * 
221      * @this {App}
222      * 
223      * @throws {requestError} service cannot be called
224      * 
225      * @example
226      * PG.App.call( {String} service )
227      * .then( {Function} callback)
228      * 
229      * @since version 1.0.0
230      */
231     App.prototype.call = function (s)
232     {
233         var promise = new PG.Promise();
234         
235         PG.Util.log('call service : ' + s);
236         
237         PG.App.rpc.request(
238             {
239                 url: PG.Cache.protocol + PG.Cache.config.url + s + '?apiKey=' + PG.Cache.settings.apiKey
240             },
241             function (rpcdata)
242             {
243                 promise.resolve(easyXDM.getJSONObject().parse(rpcdata.data));
244             },
245             function (error)
246             {
247                 throw 'Api Error, code : ' + error.code + ', msg : ' + error.message;
248             }
249         );
250         
251         return promise;
252     };
253     
254     // create App instance
255     var app = new App();
256     
257     // put the instance of App into the namespace Adfab.Playground.App
258     try {
259         addToNamespace('App', app);
260     }catch(e) {
261        throw new Error( "Cannot extends 'app' to 'Adfab.playground.App'" );
262     }
263     
264     // Let's start over there
265     app.init();
266 });
