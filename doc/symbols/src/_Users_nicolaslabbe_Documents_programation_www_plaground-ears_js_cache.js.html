 31 pl.ready(function ()
 32 {
 33     // create simple object 
 34     cache = {
 35         window: window,
 36         document: document, 
 37         xmlhttp: window.XMLHttpRequest,
 38         protocol: ('https:' == document.location.protocol ? 'https://' : 'http://'),
 39         isTouchDevice: (!!('ontouchstart' in window) || !!('onmsgesturechange' in window)),
 40         clickEvent: (!!('ontouchstart' in window) || !!('onmsgesturechange' in window)) ? 'touchstart' : 'click',
 41         config: pl_config,
 42         settings: _plgd_settings,
 43         localstorage: ((typeof window.localStorage == 'undefined' || typeof window.sessionStorage == 'undefined') ? false : true)
 44     };
 45     
 46     // put the cache into Adfab.Playground.Cache
 47     try {
 48         addToNamespace('Cache', cache);
 49     }catch(e) {
 50        throw new Error( "Cannot extends 'cache' to 'Adfab.playground.Cache'" );
 51     }
 52 });
