 32 pl.ready(function ()
 33 {
 34     'use strict';
 35     
 36     var util = {
 37         
 38         /**
 39          * Check if a variable is null
 40          * @param {Object} object
 41          * @return {Boolean} true or false
 42          * @this {Object} util
 43          */
 44         not_null: function (obj)
 45         {
 46             'use strict';
 47             
 48             return ((obj != null) ? ((typeof(obj) != 'undefined') ? ((obj !== '') ? true : false) : false) : false);
 49         },
 50         
 51         /**
 52          * Return the content of the given tag
 53          * @param {Object} DOM Object
 54          * @return {String} value
 55          * @this {Object} util
 56          */
 57         getValueFromObject: function (obj)
 58         {
 59             'use strict';
 60             var value = '';
 61             if(NS.Util.not_null(obj) && typeof obj == 'object' && obj.type != undefined) {
 62                 if(obj.value != undefined) {
 63                     value = obj.value;
 64                 }else if(obj.innerHTML != undefined) {
 65                     value = obj.innerHTML;
 66                 }
 67             }
 68             return value;
 69         },
 70         
 71         /**
 72          * Create a cookie
 73          * @param {String} name
 74          * @param {String} value
 75          * @param {Integer} number of days
 76          * @return {null} no return
 77          * @this {Object} util
 78          */
 79         createCookie: function (name, value, days)
 80         {
 81             'use strict';
 82             
 83             if (days) {
 84                 var date = new Date();
 85                 date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
 86                 var expires = "; expires=" + date.toGMTString();
 87             }
 88             else var expires = "";
 89             document.cookie = NS.Cache.config.ns + '.' + name + "=" + encodeURIComponent(value) + expires + "; path=/";
 90         },
 91         
 92         /**
 93          * Get value of a cookie
 94          * @param {String} name
 95          * @return {String} | {null} value of the cookie
 96          * @this {Object} util
 97          */
 98         readCookie: function (name)
 99         {
100             'use strict';
101             
102             var nameEQ = NS.Cache.config.ns + '.' + name + "=",
103                 ca = document.cookie.split(';'),
104                 i = 0,
105                 c = '';
106                 
107             for(i = 0; i < ca.length; i++) {
108                 c = ca[i];
109                 while (c.charAt(0)==' ') c = c.substring(1,c.length);
110                     if (c.indexOf(nameEQ) == 0) return decodeURIComponent(c.substring(nameEQ.length, c.length));
111             }
112             return null;
113         },
114         
115         /**
116          * Delete a cookie
117          * @param {String} name
118          * @return {null} no return
119          * @this {Object} util
120          */
121         eraseCookie: function (name)
122         {
123             'use strict';
124             
125             var expires = 'expires=Thu, 01 Jan 1970 00:00:01 GMT';
126             document.cookie = NS.Cache.config.ns + '.' + name + "=''; " + expires + "; path=/";
127         },
128         
129         /**
130          * Tell if input URL match current URL
131          * @param {String} url
132          * @return {Boolean} true or false
133          * @this {Object} util
134          */
135         matchUrl: function (url)
136         {
137             'use strict';
138             return (top.location.href.indexOf(url) >= 0) ? true : false;
139         },
140         
141         /**
142          * Tell if input URL is a good url to watch
143          * We need to exclude playground URL
144          * @param {String} url
145          * @return {Boolean} true or false
146          * @this {Object} util
147          */
148         isUrlValid: function (url)
149         {
150             'use strict';
151             return (NS.Cache.config.url.indexOf(url) >= 0) ? false : true;
152         },
153         
154         /**
155          * log string if config allow it 
156          * @param {Object} str
157          * @return {null} no return
158          * @this {Object} util
159          */
160         log: function (str)
161         {
162             if(NS.Cache.config) {
163                 console.log("%c> PLAYGROUND DEBUG ::", "color:blue; font-weight:bold;")
164                 console.log(str)
165             }
166         }
167     };
168     
169     // put the util into Adfab.Playground.Util
170     try {
171         addToNamespace('Util', util);
172     }catch(e) {
173        throw new Error( "Cannot extends 'Util' to 'Adfab.playground.Util'" );
174     } 
175 });
