 32 pl.ready(function ()
 33 {
 34     'use strict';
 35     
 36     var util = {
 37         
 38         /**
 39          * Check if a variable is null
 40          * @param {Object} object
 41          * @return {Boolean} true or false
 42          * @this {Object} util
 43          */
 44         not_null: function (obj)
 45         {
 46             'use strict';
 47             
 48             return ((obj != null) ? ((typeof(obj) != 'undefined') ? ((obj !== '') ? true : false) : false) : false);
 49         },
 50         
 51         /**
 52          * Return the content of the given tag
 53          * @param {Object} DOM Object
 54          * @return {String} value
 55          * @this {Object} util
 56          */
 57         getValueFromObject: function (obj)
 58         {
 59             'use strict';
 60             var value = '';
 61             if(NS.Util.not_null(obj) && typeof obj == 'object' && obj.type != undefined) {
 62                 if(obj.value != undefined) {
 63                     value = obj.value;
 64                 }else if(obj.innerHTML != undefined) {
 65                     value = obj.innerHTML;
 66                 }
 67             }
 68             return value;
 69         },
 70         
 71         /**
 72          * Create a cookie
 73          * @param {String} name
 74          * @param {String} value
 75          * @param {Integer} number of days
 76          * @return {null} no return
 77          * @this {Object} util
 78          */
 79         createCookie: function (name, value, days)
 80         {
 81             'use strict';
 82             
 83             // if localstorage enabled
 84             if(NS.Cache.localstorage) {
 85             	localStorage.setItem(NS.Cache.config.ns + '.' + name, value);
 86         	// if localstorage not enable use cookies as polyfill
 87             }else {
 88 	            if (days) {
 89 	                var date = new Date();
 90 	                date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
 91 	                var expires = "; expires=" + date.toGMTString();
 92 	            }
 93 	            else var expires = "";
 94 	            document.cookie = NS.Cache.config.ns + '.' + name + "=" + encodeURIComponent(value) + expires + "; path=/";
 95             }
 96         },
 97         
 98         /**
 99          * Get value of a cookie
100          * @param {String} name
101          * @return {String} | {null} value of the cookie
102          * @this {Object} util
103          */
104         readCookie: function (name)
105         {
106             'use strict';
107             
108             // if localstorage enabled
109             if(NS.Cache.localstorage) {
110             	return localStorage.getItem(NS.Cache.config.ns + '.' + name);
111         	// if localstorage not enable use cookies as polyfill
112             }else {
113 	            var nameEQ = NS.Cache.config.ns + '.' + name + "=",
114 	                ca = document.cookie.split(';'),
115 	                i = 0,
116 	                c = '';
117 	                
118 	            for(i = 0; i < ca.length; i++) {
119 	                c = ca[i];
120 	                while (c.charAt(0)==' ') c = c.substring(1,c.length);
121 	                    if (c.indexOf(nameEQ) == 0) return decodeURIComponent(c.substring(nameEQ.length, c.length));
122 	            }
123 			}
124             return null;
125         },
126         
127         /**
128          * Delete a cookie
129          * @param {String} name
130          * @return {null} no return
131          * @this {Object} util
132          */
133         eraseCookie: function (name)
134         {
135             'use strict';
136             
137             // if localstorage enabled
138             if(NS.Cache.localstorage) {
139             	return localStorage.removeItem(NS.Cache.config.ns + '.' + name);
140         	// if localstorage not enable use cookies as polyfill
141             }else {
142 	            var expires = 'expires=Thu, 01 Jan 1970 00:00:01 GMT';
143 	            document.cookie = NS.Cache.config.ns + '.' + name + "=''; " + expires + "; path=/";
144 			}
145         },
146         
147         /**
148          * Tell if input URL match current URL
149          * @param {String} url
150          * @return {Boolean} true or false
151          * @this {Object} util
152          */
153         matchUrl: function (url)
154         {
155             'use strict';
156             return (top.location.href.indexOf(url) >= 0) ? true : false;
157         },
158         
159         /**
160          * Tell if input URL is a good url to watch
161          * We need to exclude playground URL
162          * @param {String} url
163          * @return {Boolean} true or false
164          * @this {Object} util
165          */
166         isUrlValid: function (url)
167         {
168             'use strict';
169             return (NS.Cache.config.url.indexOf(url) >= 0) ? false : true;
170         },
171         
172         /**
173          * log string if config allow it 
174          * @param {Object} str
175          * @return {null} no return
176          * @this {Object} util
177          */
178         log: function (str)
179         {
180             if(NS.Cache.config) {
181                 console.log("%c> PLAYGROUND DEBUG ::", "color:blue; font-weight:bold;")
182                 console.log(str)
183             }
184         },
185        
186        /**
187         * get dom element from object.selector and object.name
188         * {String} "selector" and {String} "name"
189         * @param {Object} item object with variable
190         * @return {Object} dom object
191         * @this {Object} User
192         */
193        getDomElemntFromItem: function (it)
194        {
195             var obj = null,
196                 o;
197            
198             if(it.selector === 'id') {
199                 obj = document.getElementById(it.name);
200             }else if(it.selector === 'class') {
201                 obj = document.getElementsByClassName(it.name);
202                 if(obj !== null && obj.length > 0) {
203                     obj = obj[0];
204                 }
205             }
206            
207             return obj;
208        }
209     };
210     
211     // put the util into Adfab.Playground.Util
212     try {
213         addToNamespace('Util', util);
214     }catch(e) {
215        throw new Error( "Cannot extends 'Util' to 'Adfab.playground.Util'" );
216     } 
217 });
