 20 /**
 21  * create a namespace and put the destination variable
 22  * into the namespace source
 23  * 
 24  * @param {Object} destination
 25  * @param {Object} source
 26  * @return {Object} destination
 27  */
 28 function extend (destination, source)
 29 {
 30     'use strict';
 31     
 32     var toString = Object.prototype.toString,  
 33         objTest = toString.call({}),
 34         property;
 35     
 36     for (property in source) {
 37         if (source[property] && objTest == toString.call(source[property])) {  
 38             destination[property] = destination[property] || {};  
 39             extend(destination[property], source[property]);  
 40         } else {  
 41             destination[property] = source[property];  
 42         }  
 43     }
 44     
 45     return destination;  
 46 };

 48 /**
 49  * Add a variable to the namespace "XXXX.Playground"
 50  * then call extends() to create the namespace
 51  * 
 52  * @param {Object} ns
 53  * @param {Object} obj
 54  * @return {Object} destination
 55  */
 56 function addToNamespace (ns, obj)
 57 {
 58     'use strict';
 59     
 60     var Playground = {};
 61     Playground[ns] = obj;
 62     
 63     return extend (
 64         Adfab,
 65         {
 66             Playground: Playground
 67         }
 68     );  
 69 };
