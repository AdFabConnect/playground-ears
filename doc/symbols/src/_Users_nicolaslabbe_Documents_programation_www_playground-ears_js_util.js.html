 19 pl.ready(function ()
 20 {
 21     'use strict';
 22     
 23 	/**
 24 	 * @namespace reference to Adfab.Playground.Util object
 25 	 */
 26 	
 27     /** Util Object
 28      * @class
 29      * @name PG.Util
 30      */
 31     var util = {
 32         
 33         /**
 34          * Check if a variable is null
 35 	     * @function
 36          * 
 37 	     * @name PG.Util.not_null
 38 	     * 
 39          * @param {Object} object DOM object
 40          * @return {boolean} bool true or false
 41 	     * 
 42 	     * @this {Util}
 43 	     * 
 44 	     * @example
 45 	     * PG.Util.not_null( Object | String | ... )
 46 	     * 
 47 	     * @since version 1.0.0
 48          */
 49         not_null: function (obj)
 50         {
 51             'use strict';
 52             
 53             return ((obj != null) ? ((typeof(obj) != 'undefined') ? ((obj !== '') ? true : false) : false) : false);
 54         },
 55         
 56         /**
 57          * Return the content of the given tag
 58 	     * @function
 59          * 
 60 	     * @name PG.Util.getValueFromObject
 61 	     * 
 62          * @param {Object} DOM Object
 63          * @return {String} value
 64 	     * 
 65 	     * @this {Util}
 66 	     * 
 67 	     * @example
 68 	     * PG.Util.getValueFromObject( Object | String | ... )
 69 	     * 
 70 	     * @since version 1.0.0
 71          */
 72         getValueFromObject: function (obj)
 73         {
 74             'use strict';
 75             
 76             var value = '';
 77             if(PG.Util.not_null(obj) && typeof obj == 'object' && obj.type != undefined) {
 78                 if(obj.value != undefined) {
 79                     value = obj.value;
 80                 }else if(obj.innerHTML != undefined) {
 81                     value = obj.innerHTML;
 82                 }
 83             }
 84             return value;
 85         },
 86         
 87         /**
 88          * Create a cookie under Adfab.Playground.[ name ]
 89 	     * @function
 90 	     * 
 91 	     * @name PG.Util.createCookie
 92 	     * 
 93          * @param {String} name
 94          * @param {String} value
 95          * @param {Number} [number] of days (optional)
 96          * @return {null} no return
 97          * 
 98 	     * @this {Util}
 99 	     * 
100 	     * @example
101 	     * PG.Util.createCookie( {String} name, {String} value, [ {Number} days ])
102 	     * 
103 	     * @since version 1.0.0
104          */
105         createCookie: function (name, value, days)
106         {
107             'use strict';
108             
109             // if localstorage enabled
110             if(PG.Cache.localstorage) {
111             	localStorage.setItem(PG.Cache.config.ns + '.' + name, value);
112         	// if localstorage not enable use cookies as polyfill
113             }else {
114 	            if (days) {
115 	                var date = new Date();
116 	                date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
117 	                var expires = "; expires=" + date.toGMTString();
118 	            }
119 	            else var expires = "";
120 	            document.cookie = PG.Cache.config.ns + '.' + name + "=" + encodeURIComponent(value) + expires + "; path=/";
121             }
122         },
123         
124         /**
125          * Get value of a cookie
126 	     * @function
127 	     * 
128 	     * @name PG.Util.readCookie
129 	     * 
130          * @param {String} name
131          * @return {String} | {null} value of the cookie
132          * 
133 	     * @this {Util}
134 	     * 
135 	     * @example
136 	     * PG.Util.readCookie( {String} name)
137 	     * 
138 	     * @since version 1.0.0
139          */
140         readCookie: function (name)
141         {
142             'use strict';
143             
144             // if localstorage enabled
145             if(PG.Cache.localstorage) {
146             	return localStorage.getItem(PG.Cache.config.ns + '.' + name);
147         	// if localstorage not enable use cookies as polyfill
148             }else {
149 	            var nameEQ = PG.Cache.config.ns + '.' + name + "=",
150 	                ca = document.cookie.split(';'),
151 	                i = 0,
152 	                c = '';
153 	                
154 	            for(i = 0; i < ca.length; i++) {
155 	                c = ca[i];
156 	                while (c.charAt(0)==' ') c = c.substring(1,c.length);
157 	                    if (c.indexOf(nameEQ) == 0) return decodeURIComponent(c.substring(nameEQ.length, c.length));
158 	            }
159 			}
160             return null;
161         },
162         
163         /**
164          * Delete a cookie
165 	     * @function
166 	     * 
167 	     * @name PG.Util.eraseCookie
168 	     * 
169          * @param {String} name
170          * @return {null} no return
171          * 
172 	     * @this {Util}
173 	     * 
174 	     * @example
175 	     * PG.Util.readCookie( {String} name)
176 	     * 
177 	     * @since version 1.0.0
178          */
179         eraseCookie: function (name)
180         {
181             'use strict';
182             
183             // if localstorage enabled
184             if(PG.Cache.localstorage) {
185             	return localStorage.removeItem(PG.Cache.config.ns + '.' + name);
186         	// if localstorage not enable use cookies as polyfill
187             }else {
188 	            var expires = 'expires=Thu, 01 Jan 1970 00:00:01 GMT';
189 	            document.cookie = PG.Cache.config.ns + '.' + name + "=''; " + expires + "; path=/";
190 			}
191         },
192         
193         /**
194          * Tell if input URL match current URL
195 	     * @function
196 	     * 
197 	     * @name PG.Util.matchUrl
198 	     * 
199          * @param {String} url
200          * @return {Boolean} bool true or false
201          * 
202 	     * @this {Util}
203 	     * 
204 	     * @example
205 	     * PG.Util.matchUrl( {String} url)
206 	     * 
207 	     * @since version 1.0.0
208          */
209         matchUrl: function (url)
210         {
211             'use strict';
212             return (top.location.href.indexOf(url) >= 0) ? true : false;
213         },
214         
215         /**
216          * Tell if input URL is a good url to watch
217          * We need to exclude playground URL
218 	     * @function
219 	     * 
220          * @param {String} url
221          * @return {Boolean} bool true or false
222          * 
223 	     * @this {Util}
224 	     * 
225 	     * @example
226 	     * PG.Util.isUrlValid( {String} url)
227 	     * 
228 	     * @since version 1.0.0
229          */
230         isUrlValid: function (url)
231         {
232             'use strict';
233             return (PG.Cache.config.url.indexOf(url) >= 0) ? false : true;
234         },
235         
236         /**
237          * log string if PG.Cache.Config.debug allow it
238 	     * @function
239 	     * 
240 	     * @name PG.Util.log
241 	     * 
242          * @param {Object} str
243          * @return {null}
244          * 
245 	     * @this {Util}
246 	     * 
247 	     * @example
248 	     * PG.Util.log( {String} str)
249 	     * 
250 	     * @since version 1.0.0
251          */
252         log: function (str)
253         {
254             if(PG.Cache.config.debug) {
255                 console.log("%c> PLAYGROUND DEBUG ::", "color:blue; font-weight:bold;")
256                 console.log(str)
257             }
258         },
259        
260        /**
261         * get dom element from object.selector and object.name
262 	    * @function
263 	    * 
264 	    * @name PG.Util.getDomElemntFromItem
265 	    * 
266         * @param {Object} item object with variable
267         * @param item.selector type of the DOM element
268         * @param item.name id or class of the DOM element
269         * @return {Object} obj DOM object
270         * 
271 	    * @this {Util}
272 	    * 
273 	    * @example
274 	    * PG.Util.getDomElemntFromItem( {String} it)
275 	    * 
276 	    * @since version 1.0.0
277         */
278        getDomElemntFromItem: function (it)
279        {
280             var obj = null,
281                 o;
282            
283             if(it.selector === 'id') {
284                 obj = document.getElementById(it.name);
285             }else if(it.selector === 'class') {
286                 obj = document.getElementsByClassName(it.name);
287                 if(obj !== null && obj.length > 0) {
288                     obj = obj[0];
289                 }
290             }
291            
292             return obj;
293        }
294     };

297     try {
298         addToNamespace('Util', util);
299     }catch(e) {
300        throw new Error( "Cannot extends 'Util' to 'Adfab.playground.Util'" );
301     }
