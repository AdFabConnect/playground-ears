 22 pl.ready(function ()
 23 {
 24     'use strict';
 25     
 26     var user = {
 27         
 28         env: { },
 29         id: null,
 30         uid: null,
 31         data: { },
 32         urls: {
 33             current: top.location.href,
 34             prev: null
 35         },
 36         
 37         /**
 38          * @contructor
 39          * @param {null} no param
 40          * @return {null} no return
 41          * @this {Object} User
 42          */ 
 43         init: function ()
 44         {
 45             NS.User.GenerateUniqueId();
 46             NS.User.urls.prev = NS.Util.readCookie('prev-url');
 47             
 48             var authent = NS.Util.readCookie('authent');
 49             if(!NS.Util.not_null(authent)) {
 50             	NS.User.loadAuthent()
 51             	.then(function ()
 52             	{
 53                 	NS.User.checkUser();
 54             	});
 55             }else {
 56                 NS.User.data = JSON.parse(NS.Util.readCookie('authent'));
 57             	NS.User.checkUser();
 58             }
 59         },
 60         
 61         /**
 62          * check on cookie if user is logged
 63          * @param {null} no param
 64          * @return {Boolean} true or false
 65          */
 66         isLogged: function ()
 67         {
 68             var l;
 69             if(NS.User.id === '' || NS.User.id === null) {
 70                 // check
 71                 NS.User.id = NS.Util.readCookie('login');
 72             }
 73             l = (NS.Util.not_null(NS.User.id)) ? true : false;
 74             NS.Util.log('isLogged : ' + l);
 75             return l;
 76         },
 77         
 78         /**
 79          * Call service to get json data of logout user method
 80          * put the return into NS.User.data
 81          * then call bindLogout
 82          * @param {null} no param
 83          * @return {null} no return
 84          * @this {Object} User
 85          */
 86         loadAuthent: function ()
 87         {
 88         	var p = new NS.Promise();
 89         	
 90             NS.App.call('connect.php')
 91             .then(
 92                 function (data)
 93                 {
 94                     NS.User.data = data;
 95                     NS.Util.createCookie('authent', JSON.stringify(data));
 96                 	p.resolve();
 97                     //NS.User.bindLogout();
 98                 }
 99             );
100             
101             return p;
102         },
103         
104         checkUser: function ()
105         {
106             if(NS.Util.not_null(NS.Util.readCookie('logout-try'))) { // USER TRIED TO LOUGOUT
107                 //NS.User.bindLogout();
108                 if(NS.User.checkEvidences(NS.User.data.logout.evidences)) {
109                 	// IS OUT
110                 	NS.User.logout();
111                 }
112             }else if(NS.Util.not_null(NS.Util.readCookie('login-try'))) { // USER TRIED TO LOGIN
113                 //NS.User.bindLogin();
114                 if(NS.User.checkEvidences(NS.User.data.login.evidences)) {
115                 	// IS IN
116                 	NS.User.login(NS.Util.readCookie('login-try'));
117                 }else {
118                 	NS.Util.eraseCookie('login-try')
119                 }
120             }else if(NS.Util.not_null(NS.Util.readCookie('login'))) { // USER IS ALREADY LOGGED
121                 //NS.User.bindLogout();
122                 NS.User.login(NS.Util.readCookie('login'));
123             }else { // USER IS LOGGED OUT
124                 NS.User.logout();
125             }
126         },
127          
128         /**
129          * Set the login for current user if param {String} not null
130          * call loadLogout to track logout since the user is logged
131          * then return the login
132          * @param {String} str (optional) if not set juste return the login
133          * @return {String} NS.User.data.login
134          * @this {Object} User
135          */
136         login: function (str)
137         {
138             NS.Util.log('Login "' + str + '"');
139             if(str !== null) {
140             	NS.Util.eraseCookie('login-try')
141                 NS.User.id = str;
142                 NS.Util.createCookie('login', str); 
143                 
144                 // so bind logout
145                 if(!NS.Util.not_null(NS.User.data.urls)) {
146                     //NS.User.loadLogout();
147                 }
148                 NS.App.send(top.location.href);
149             }
150             
151             return NS.User.data.id;
152         },
153         
154         /**
155          * force logout user, then call loadLogin since user is logged out
156          * @param {null} no param
157          * @return {null} no return
158          * @this {Object} User
159          */
160         logout: function ()
161         {
162             NS.Util.log('User is out');
163             NS.Util.eraseCookie('logout-try');
164             NS.Util.eraseCookie('login');
165             NS.User.id = null;
166             NS.App.send(top.location.href);
167             
168             // so bind login
169             // NS.User.loadLogin();
170         },
171         
172         /**
173          * Generate unique id for anonymous user
174          * @param {null} no param
175          * @return {String} the id
176          * @this {Object} User
177          */
178         GenerateUniqueId: function ()
179         {
180             var uui = NS.Util.readCookie('unique-id');
181             if(!NS.Util.not_null(uui)) {
182                 uui = 'body_' + new Date().getTime();
183                 NS.Util.createCookie('unique-id', uui);
184             }
185             NS.User.uid = uui;
186             return uui;
187         },
188         
189         /**
190          * Check evidences for current user.evidences object
191          * @param {null} no param
192          * @return {Boolean} result of the checking
193          * @this {Object} User
194          */
195         checkEvidences: function (evidences)
196         {
197             var obj = null,
198                 evidencesOk = true,
199                 it;
200             
201             for(it in evidences) {
202                 if(evidences[it].selector === 'id') {
203                     obj = document.getElementById(evidences[it].name);
204                 }else if(NS.User.data.evidences[it].selector === 'class') {
205                     obj = document.getElementsByClassName(evidences[it].name);
206                 }
207                 if(obj === null) {
208                     evidencesOk = false;
209                 }
210             }
211             
212             return evidencesOk;
213         },
214         
215         /**
216          * Call this method when user quit the current to check if try to logout/login
217          * make cookies 'login-try' and 'logout-try'
218          * @param {null} no param
219          * @return {null} no return
220          * @this {Object} User
221          */
222         quit: function ()
223         { 
224             var it,
225                 id = '';
226             
227             NS.Util.createCookie('prev-url', top.location.href);
228         	
229             if(NS.Util.not_null(NS.User.data.login) && NS.Util.not_null(NS.User.data.logout)) {
230            		// check user is on login page 
231             	if(!NS.User.isLogged() && NS.Util.matchUrl(NS.User.data.login.urls.page)) {
232                     for(it in NS.User.data.login.items) {
233                         id += (id !== '') ? ':' : '';
234                         id += NS.Util.getValueFromObject(
235                             NS.Util.getDomElemntFromItem(NS.User.data.login.items[it])
236                         );
237                     }
238                 	NS.Util.createCookie('login-try', id);
239            	 	// check user is on logout page
240                 }else if(NS.Util.matchUrl(NS.User.data.logout.urls.page)) {
241                 	NS.Util.createCookie('logout-try', 'logout');
242                 }
243             }
244         }
245     };
246     
247     // put the user into Adfab.Playground.User
248     try {
249         addToNamespace('User', user);
250     }catch(e) {
251        throw new Error( "Cannot extends 'User' to 'Adfab.playground.User'" );
252     }
253 });
