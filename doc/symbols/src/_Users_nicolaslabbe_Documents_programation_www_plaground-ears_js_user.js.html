 18 pl.ready(function ()
 19 {
 20     'use strict';
 21     
 22     /** User Object
 23      * @class
 24      * @name NS.User
 25      */
 26     var user = {
 27         
 28         env: { },
 29         id: null,
 30         uid: null,
 31         data: { },
 32         urls: {
 33             current: top.location.href,
 34             prev: null
 35         },
 36         
 37         /**
 38          * Init the user object
 39          * @function
 40 	     * 
 41 	     * @name NS.User.init
 42 	     * 
 43          * @param {null}
 44          * @return {null}
 45 	     * 
 46 	     * @this {User}
 47 	     * 
 48 	     * @ignore
 49 	     * 
 50 	     * @since version 1.0.0
 51          */ 
 52         init: function ()
 53         {
 54             NS.User.GenerateUniqueId();
 55             NS.User.urls.prev = NS.Util.readCookie('prev-url');
 56             
 57             var authent = NS.Util.readCookie('authent');
 58             if(!NS.Util.not_null(authent)) {
 59             	NS.User.loadAuthent()
 60             	.then(function ()
 61             	{
 62                 	NS.User.checkUser();
 63             	});
 64             }else {
 65                 NS.User.data = JSON.parse(NS.Util.readCookie('authent'));
 66             	NS.User.checkUser();
 67             }
 68         },
 69         
 70         /**
 71          * check on cookie if user is logged
 72          * @function
 73 	     * 
 74 	     * @name NS.User.isLogged
 75 	     * 
 76          * @param {null} 
 77          * @return {Boolean} bool true or false
 78 	     * 
 79 	     * @this {User}
 80 	     * 
 81 	     * @example
 82 	     * NS.User.isLogged()
 83 	     * 
 84 	     * @since version 1.0.0
 85          */
 86         isLogged: function ()
 87         {
 88             var l;
 89             if(NS.User.id === '' || NS.User.id === null) {
 90                 // check
 91                 NS.User.id = NS.Util.readCookie('login');
 92             }
 93             l = (NS.Util.not_null(NS.User.id)) ? true : false;
 94             NS.Util.log('isLogged : ' + l);
 95             return l;
 96         },
 97         
 98         /**
 99          * Call service to get json data of logout user method
100          * put the return into NS.User.data
101          * then call bindLogout
102          * @function
103 	     * 
104 	     * @name NS.User.loadAuthent
105 	     * 
106          * @param {null}
107          * @return {Object} promise
108 	     * 
109 	     * @this {User}
110 	     * 
111 	     * @ignore
112 	     * 
113 	     * @since version 1.0.0
114          */
115         loadAuthent: function ()
116         {
117         	var p = new NS.Promise();
118         	
119             NS.App.call('connect.php')
120             .then(
121                 function (data)
122                 {
123                     NS.User.data = data;
124                     NS.Util.createCookie('authent', JSON.stringify(data));
125                 	p.resolve();
126                 }
127             );
128             
129             return p;
130         },
131         
132         /**
133          * Check if the user is logged, logout, or tried to login/logout 
134          * @function
135 	     * 
136 	     * @name NS.User.checkUser
137 	     * 
138          * @param {null}
139          * @return {null}
140 	     * 
141 	     * @this {User}
142 	     * 
143 	     * @ignore
144 	     * 
145 	     * @since version 1.0.0
146          */
147         checkUser: function ()
148         {
149             if(NS.Util.not_null(NS.Util.readCookie('logout-try'))) { // USER TRIED TO LOUGOUT
150                 //NS.User.bindLogout();
151                 if(NS.User.checkEvidences(NS.User.data.logout.evidences)) {
152                 	// IS OUT
153                 	NS.User.logout();
154                 }
155             }else if(NS.Util.not_null(NS.Util.readCookie('login-try'))) { // USER TRIED TO LOGIN
156                 //NS.User.bindLogin();
157                 if(NS.User.checkEvidences(NS.User.data.login.evidences)) {
158                 	// IS IN
159                 	NS.User.login(NS.Util.readCookie('login-try'));
160                 }else {
161                 	NS.Util.eraseCookie('login-try')
162                 }
163             }else if(NS.Util.not_null(NS.Util.readCookie('login'))) { // USER IS ALREADY LOGGED
164                 //NS.User.bindLogout();
165                 NS.User.login(NS.Util.readCookie('login'));
166             }else { // USER IS LOGGED OUT
167                 NS.User.logout();
168             }
169         },
170          
171         /**
172          * Set the login for current user if param {String} not null
173          * call loadLogout to track logout since the user is logged
174          * then return the login
175          * @function
176 	     * 
177 	     * @name NS.User.login
178 	     * 
179          * @param {String} str (optional) if not set just return the login
180          * @return {String} NS.User.data.id
181 	     * 
182 	     * @this {User}
183 	     * 
184 	     * @example
185 	     * NS.User.login( {String} username )
186 	     * 
187 	     * @since version 1.0.0
188          */
189         login: function (str)
190         {
191             NS.Util.log('Login "' + str + '"');
192             if(str !== null) {
193             	NS.Util.eraseCookie('login-try')
194                 NS.User.id = str;
195                 NS.Util.createCookie('login', str); 
196                 
197                 // so bind logout
198                 if(!NS.Util.not_null(NS.User.data.urls)) {
199                     //NS.User.loadLogout();
200                 }
201                 NS.App.send(top.location.href);
202             }
203             
204             return NS.User.data.id;
205         },
206         
207         /**
208          * force logout user, then call loadLogin since user is logged out
209          * @function
210 	     * 
211 	     * @name NS.User.logout
212 	     * 
213          * @param {null} 
214          * @return {null}
215 	     * 
216 	     * @this {User}
217 	     * 
218 	     * @example
219 	     * NS.User.logout()
220 	     * 
221 	     * @since version 1.0.0
222          */
223         logout: function ()
224         {
225             NS.Util.log('User is out');
226             NS.Util.eraseCookie('logout-try');
227             NS.Util.eraseCookie('login');
228             NS.User.id = null;
229             NS.App.send(top.location.href);
230         },
231         
232         /**
233          * Generate unique id for anonymous user
234          * @function
235          * 
236 	     * @name NS.User.GenerateUniqueId
237 	     * 
238          * @param {null}
239          * @return {String} id
240 	     * 
241 	     * @this {User}
242 	     * 
243 	     * @example
244 	     * NS.User.GenerateUniqueId()
245 	     * 
246 	     * @since version 1.0.0
247          */
248         GenerateUniqueId: function ()
249         {
250             var uui = NS.Util.readCookie('unique-id');
251             if(!NS.Util.not_null(uui)) {
252                 uui = 'body_' + new Date().getTime();
253                 NS.Util.createCookie('unique-id', uui);
254             }
255             NS.User.uid = uui;
256             return uui;
257         },
258         
259         /**
260          * Check evidences for current user.evidences object
261          * @function
262          * 
263 	     * @name NS.User.checkEvidences
264 	     * 
265          * @param {null}
266          * @return {Boolean} result true | false
267 	     * 
268 	     * @this {User}
269 	     * 
270 	     * @ignore
271 	     * 
272 	     * @since version 1.0.0
273          */
274         checkEvidences: function (evidences)
275         {
276             var obj = null,
277                 evidencesOk = true,
278                 it;
279             
280             for(it in evidences) {
281                 if(evidences[it].selector === 'id') {
282                     obj = document.getElementById(evidences[it].name);
283                 }else if(NS.User.data.evidences[it].selector === 'class') {
284                     obj = document.getElementsByClassName(evidences[it].name);
285                 }
286                 if(obj === null) {
287                     evidencesOk = false;
288                 }
289             }
290             
291             return evidencesOk;
292         },
293         
294         /**
295          * Call this method when user quit the current to check if try to logout/login
296          * make cookies 'login-try' and 'logout-try'
297          * @function
298          * 
299 	     * @name NS.User.quit
300 	     * 
301          * @param {null}
302          * @return {null}
303 	     * 
304 	     * @this {User}
305 	     * 
306 	     * @ignore
307 	     * 
308 	     * @since version 1.0.0
309          */
310         quit: function ()
311         { 
312             var it,
313                 id = '';
314             
315             NS.Util.createCookie('prev-url', top.location.href);
316         	
317             if(NS.Util.not_null(NS.User.data.login) && NS.Util.not_null(NS.User.data.logout)) {
318            		// check user is on login page 
319             	if(!NS.User.isLogged() && NS.Util.matchUrl(NS.User.data.login.urls.page)) {
320                     for(it in NS.User.data.login.items) {
321                         id += (id !== '') ? ':' : '';
322                         id += NS.Util.getValueFromObject(
323                             NS.Util.getDomElemntFromItem(NS.User.data.login.items[it])
324                         );
325                     }
326                 	NS.Util.createCookie('login-try', id);
327            	 	// check user is on logout page
328                 }else if(NS.Util.matchUrl(NS.User.data.logout.urls.page)) {
329                 	NS.Util.createCookie('logout-try', 'logout');
330                 }
331             }
332         }
333     };
334     
335     // put the user into Adfab.Playground.User
336     try {
337         addToNamespace('User', user);
338     }catch(e) {
339        throw new Error( "Cannot extends 'User' to 'Adfab.playground.User'" );
340     }
341 });
